function convertPropsWithStartEnd(ruleProp) {

function convertValueWithStartEnd(ruleProp, ruleValue) {
    switch (ruleProp) {
        case 'clear':

            if (ruleValue === 'start') {

function convertPropWith4Values(ruleProp, ruleValue) {
